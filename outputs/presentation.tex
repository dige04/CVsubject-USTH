% SLIDE 8: CEILING EFFECT (1.5 min)

\begin{frame}{Key Finding: The Ceiling Effect}
    \centering
    \vspace{0.3em}

    {\large Both modalities \textbf{independently} saturate accuracy.\\[0.2em]
    Fusion adds \textbf{zero measurable benefit}.}

    \vspace{1em}
    \begin{columns}[T]
    \begin{column}{0.45\textwidth}
        \textbf{Why does this happen?}
        \begin{enumerate}
            \setlength\itemsep{0.3em}
            \item 5 gestures are \textbf{geometrically distinct}\\{\scriptsize (fist $\neq$ open hand $\neq$ pinch)}
            \item Person-aware CV: no identity leakage
            \item 13,250 subjects $\to$ robust generalization
        \end{enumerate}
    \end{column}
    \begin{column}{0.45\textwidth}
        \textbf{Implication:}
        \begin{enumerate}
            \setlength\itemsep{0.3em}
            \item \textbf{Task difficulty} drives accuracy,\\not model capacity
            \item For simple vocabularies:\\a tiny MLP suffices
            \item Fusion helps at harder tasks\\{\scriptsize (full HaGRID: 18 classes, SOTA 95.5\%)}
        \end{enumerate}
    \end{column}
    \end{columns}

    \vspace{0.8em}
    \begin{block}{}
        \centering
        \textbf{Answer to Q1 \& Q2:} Deploy a 227 KB MLP on skeleton. Fusion unnecessary.
    \end{block}
\end{frame}

\begin{frame}{Mathematical Interaction Model (adapted from Huda et al. 2025)}
    \vspace{0.3em}
    \begin{columns}[c]
    \begin{column}{0.42\textwidth}
        \centering
        \includegraphics[width=\textwidth,height=5cm,keepaspectratio]{interaction_model.png}

        \vspace{0.3em}
        {\scriptsize Huda: wheelchair steering\\Ours: puzzle tile swap}
    \end{column}
    \begin{column}{0.54\textwidth}
        \small
        Let $\Delta x = x_{cur} - x_{init}$,\enspace $\Delta y = y_{cur} - y_{init}$

        \vspace{0.4em}
        \begin{tabular}{ll}
        \toprule
        \textbf{Game Action} & \textbf{Condition} \\
        \midrule
        Swap Right & $\Delta x > mov_x$ AND $|\Delta y| < tol$ \\
        Swap Left & $\Delta x < -mov_x$ AND $|\Delta y| < tol$ \\
        Swap Up & $\Delta y < -mov_y$ AND $|\Delta x| < tol$ \\
        Swap Down & $\Delta y > mov_y$ AND $|\Delta x| < tol$ \\
        \bottomrule
        \end{tabular}

        \vspace{0.6em}
        \begin{tabular}{ll}
        $mov_{x,y}$ & $= 0.6 \times \text{tile\_size}$ \\
        $tol$ & $= 0.3 \times \text{tile\_size}$ \\
        \end{tabular}

        \vspace{0.5em}
        {\footnotesize
        \textbf{Key design:}\\
        $\bullet$ Origin resets after each swap $\to$ multi-step swiping\\
        $\bullet$ 3-frame debounce filters transient errors\\
        $\bullet$ Thresholds tied to tile geometry, not pixels}
    \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Conclusion}
    \vspace{0.5em}
    \begin{columns}[T]
    \begin{column}{0.55\textwidth}
        \textbf{Contributions:}
        \begin{enumerate}
            \setlength\itemsep{0.8em}
            \item \textbf{Multi-modal finding:} Skeleton (99.8\%) and appearance (100\%) independently saturate $\to$ \textbf{ceiling effect}. Fusion unnecessary for 5 coarse gestures.

            \item \textbf{Interaction model:} Adapted Huda et al.'s directional thresholds ($mov_x$, $mov_y$, $tol$) from wheelchair control $\to$ puzzle tile swiping.

            \item \textbf{Deployed:} 227 KB MLP, 100\% client-side, 30+ FPS, no server.
        \end{enumerate}
    \end{column}
    \begin{column}{0.4\textwidth}
        \textbf{Limitations:}
        \begin{itemize}
            \item 5 gestures are trivially separable
            \item No formal user study
            \item Fixed thresholds (not per-user)
        \end{itemize}

        \vspace{0.8em}
        \textbf{Future work:}
        \begin{itemize}
            \item Full HaGRID (18 classes)\\where fusion matters
            \item Adaptive threshold calibration
            \item Formal usability evaluation
        \end{itemize}
    \end{column}
    \end{columns}
\end{frame}
