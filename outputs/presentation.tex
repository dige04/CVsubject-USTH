    \vspace{1em}
    {\Large \textbf{The Core Finding:}}\\
    \vspace{0.5em}
    The 5 discrete classes are simply \textbf{too easy} for modern architectures.
    
    \vspace{1em}
    \begin{itemize}
        \setlength\itemsep{0.8em}
        \item A tiny \textbf{14K-parameter MLP} essentially saturates the dataset (99.8\%).
        \item \textbf{Fusion offered zero value}: Geometric features alone perfectly resolve spatial ambiguity.
        \item \textbf{Winner:} MLP on MediaPipe landmarks provides the ultimate lightweight engine.
    \end{itemize}
\end{column}
\end{columns}

% ======== 5. CLASSIFICATION != CONTROL ========
\begin{frame}{The Real Challenge: Classification $\neq$ Control}
    \centering
    \vspace{1.5em}
    
    {\Large We solved classification (99.8\% accuracy).\\ But feeding raw frames directly to the game creates a terrible UX.}
    
    \vspace{2.5em}
    \begin{columns}[T]
        \begin{column}{0.3\textwidth}
            \centering
            \textbf{Hand Jitter}\\
            {\small Natural micro-movements cause erratic cursor jumps.}
        \end{column}
        \begin{column}{0.3\textwidth}
            \centering
            \textbf{Transient Errors}\\
            {\small A single misclassified frame drops the puzzle piece.}
        \end{column}
        \begin{column}{0.3\textwidth}
            \centering
            \textbf{Lack of Intent}\\
            {\small Users cannot hold perfectly still in 3D space.}
        \end{column}
    \end{columns}
    
    \vspace{3em}
    \begin{block}{}
        \centering
        \textit{"In order to make [the system] more suitable for disabled people... it needs to be more user-flexible... requires less energy" -- Huda et al. 2025}
    \end{block}
\end{frame}


% ======== 6. GRID-BASED SWIPE MATH ========
\begin{frame}{Grid-Based Swipe Math (Huda et al. 2025)}
    \centering
    \vspace{0.3em}

    {\small Adapted the discrete directional \textbf{mathematical model} from wheelchair control\\
    to filter spatial noise and enable robust, low-energy puzzle swiping.}

    \vspace{0.8em}
    \begin{columns}[c]
    \begin{column}{0.48\textwidth}
        \centering
        \begin{tikzpicture}[
            box/.style={draw=none, fill=gray!10, rounded corners, minimum width=2.8cm, minimum height=2.8cm},
            tol/.style={draw=accent, thick, dashed, fill=accent!10},
            axis/.style={-latex, thick, gray},
            move/.style={-latex, very thick, primary},
        ]
            \node[box] (bg) at (0,0) {};
            
            % Tolerance zone
            \draw[tol] (-0.8, -1.2) rectangle (0.8, 1.2);
            \node[accent, font=\scriptsize] at (0, 0.9) {tol = 0.3w};
            
            % Axes
            \draw[axis] (-1.5, 0) -- (1.5, 0) node[right] {$x$};
            \draw[axis] (0, -1.5) -- (0, 1.5) node[above] {$y$};
            
            % Initial point
            \fill (0,0) circle (3pt) node[below right, font=\footnotesize] {$(x_{init}, y_{init})$};
            
            % Movement arrows
            \draw[move] (0,0) -- (1.2, 0) node[midway, below, font=\scriptsize] {$mov_x$};
            \fill[primary] (1.3, 0.2) circle (2pt) node[above=1pt, font=\scriptsize] {$(x_{cur}, y_{cur})$};
        \end{tikzpicture}
    \end{column}
    \begin{column}{0.48\textwidth}
        \small
        \textbf{Logical Conditions}:\\ Let $\Delta x = x_{cur} - x_{init}$, $\Delta y = y_{cur} - y_{init}$
        
        \vspace{0.6em}
        \textbf{Swipe Right (swap $\to$):} \\
        $\Delta x > \mathbf{mov_x} \text{ AND } |\Delta y| < \mathbf{tol}$
        
        \vspace{0.6em}
        \textbf{Swipe Up (swap $\uparrow$):} \\
        $\Delta y < -\mathbf{mov_y} \text{ AND } |\Delta x| < \mathbf{tol}$
        
        \vspace{0.6em}
        \textit{\footnotesize *Origin updates dynamically after swap for continuous multi-step swiping.}
    \end{column}
    \end{columns}
\end{frame}
